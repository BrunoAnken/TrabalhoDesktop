-- Table: public.desenvolvedoras

-- DROP TABLE public.desenvolvedoras;

CREATE TABLE public.desenvolvedoras
(
  id integer NOT NULL DEFAULT nextval('desenvolvedoras_id_seq'::regclass),
  name character varying(100),
  origin character varying(100),
  foundation character varying(100),
  bibliotecas character varying(500),
  frameworks character varying(500)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.desenvolvedoras
  OWNER TO postgres;
